import numpy as np
import matplotlib.pyplot as plt

# 1. 生成单元数n（2的幂次：4,8,16,32,64,128,256...）
ns = [4, 8, 16, 32, 64, 128, 256, 512, 1024]
pi_exact = np.pi
pis_approx = []
errors = []

# 2. 计算每个n对应的近似π和误差
for n in ns:
    pi_approx = n * np.sin(np.pi / n)
    pis_approx.append(pi_approx)
    err = abs(pi_exact - pi_approx)
    errors.append(err)
    print(f"n={n:4d}, 近似π={pi_approx:.15f}, 误差={err:.2e}")

# 3. 计算单元尺寸h，做双对数图
hs = [1/n for n in ns]
log_h = np.log10(hs)
log_err = np.log10(errors)

# 4. 拟合求收敛阶（最小二乘求斜率）
slope, _ = np.polyfit(log_h, log_err, 1)

# 5. 绘图
plt.figure(figsize=(8,6))
plt.loglog(hs, errors, 'o-', label=f'拟合收敛阶 slope={slope:.2f}')
plt.xlabel('单元尺寸 h=1/n')
plt.ylabel('误差 |π−πₙ|')
plt.title('正n边形逼近圆周求π——双对数误差收敛曲线')
plt.legend()
plt.grid(True, which="both", ls="--")
plt.show()
